# =============================================================================

# -----------------------------------------------------------------------------
# 1. API Framework

# -----------------------------------------------------------------------------

# -----------------------------------------------------------------------------

# -----------------------------------------------------------------------------

botocore==1.35.71

# -----------------------------------------------------------------------------
